For Raspberry

    # Create and enable swap area

    sudo fallocate -l 500M $HOME/swapfile
    sudo chmod 600 $HOME/swapfile
    sudo mkswap $HOME/swapfile
    sudo swapon $HOME/swapfile


    # Remove swap area

    sudo swapoff $HOME/swapfile
    sudo rm -f $HOME/swapfile
